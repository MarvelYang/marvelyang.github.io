class KthLargest {
    constructor(k, nums) {
        this.q = new MinPriorityQueue(k);
        this.k = k;

        for (let n of nums) {
            this.add(n);
        }
    }

    add(val) {
        if (this.q.size() < k) {
            q.offer(val);
        } else if (this.q.peek() < val) {
            q.poll();
            q.offer(val);
        }

        return q.peek();
    }
}
